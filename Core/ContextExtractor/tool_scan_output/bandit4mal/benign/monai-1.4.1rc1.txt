454	def get_keywords():
455	    """Get the keywords needed to look up the version information."""
456	    # these strings will be replaced by git during git-archive.
457	    # setup.py/versioneer.py will grep for the variable names, so they must
458	    # each be defined on a line of their own. _version.py will just call
459	    # get_keywords().
460	    git_refnames = "%(DOLLAR)sFormat:%%d%(DOLLAR)s"
461	    git_full = "%(DOLLAR)sFormat:%%H%(DOLLAR)s"
462	    git_date = "%(DOLLAR)sFormat:%%ci%(DOLLAR)s"
463	    keywords = {"refnames": git_refnames, "full": git_full, "date": git_date}
464	    return keywords

493	def register_vcs_handler(vcs, method):  # decorator
494	    """Create decorator to mark a method as the handler of a VCS."""
495	    def decorate(f):
496	        """Store f in HANDLERS[vcs][method]."""
497	        if vcs not in HANDLERS:
498	            HANDLERS[vcs] = {}
499	        HANDLERS[vcs][method] = f
500	        return f
501	    return decorate

504	def run_command(commands, args, cwd=None, verbose=False, hide_stderr=False,
505	                env=None):
506	    """Call the given command(s)."""
507	    assert isinstance(commands, list)
508	    process = None
509	
510	    popen_kwargs = {}
511	    if sys.platform == "win32":
512	        # This hides the console window if pythonw.exe is used
513	        startupinfo = subprocess.STARTUPINFO()
514	        startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW
515	        popen_kwargs["startupinfo"] = startupinfo
516	
517	    for command in commands:
518	        try:
519	            dispcmd = str([command] + args)
520	            # remember shell=False, so use git.cmd on windows, not just git
521	            process = subprocess.Popen([command] + args, cwd=cwd, env=env,
522	                                       stdout=subprocess.PIPE,
523	                                       stderr=(subprocess.PIPE if hide_stderr
524	                                               else None), **popen_kwargs)
525	            break
526	        except OSError:
527	            e = sys.exc_info()[1]
528	            if e.errno == errno.ENOENT:
529	                continue
530	            if verbose:
531	                print("unable to run %%s" %% dispcmd)
532	                print(e)
533	            return None, None
534	    else:
535	        if verbose:
536	            print("unable to find command, tried %%s" %% (commands,))
537	        return None, None
538	    stdout = process.communicate()[0].strip().decode()
539	    if process.returncode != 0:
540	        if verbose:
541	            print("unable to run %%s (error)" %% dispcmd)
542	            print("stdout was %%s" %% stdout)
543	        return None, process.returncode
544	    return stdout, process.returncode

547	def versions_from_parentdir(parentdir_prefix, root, verbose):
548	    """Try to determine the version from the parent directory name.
549	
550	    Source tarballs conventionally unpack into a directory that includes both
551	    the project name and a version string. We will also support searching up
552	    two directory levels for an appropriately named parent directory
553	    """
554	    rootdirs = []
555	
556	    for _ in range(3):
557	        dirname = os.path.basename(root)
558	        if dirname.startswith(parentdir_prefix):
559	            return {"version": dirname[len(parentdir_prefix):],
560	                    "full-revisionid": None,
561	                    "dirty": False, "error": None, "date": None}
562	        rootdirs.append(root)
563	        root = os.path.dirname(root)  # up a level
564	
565	    if verbose:
566	        print("Tried directories %%s but none started with prefix %%s" %%
567	              (str(rootdirs), parentdir_prefix))
568	    raise NotThisMethod("rootdir doesn't start with parentdir_prefix")

571	@register_vcs_handler("git", "get_keywords")
572	def git_get_keywords(versionfile_abs):
573	    """Extract version information from the given file."""
574	    # the code embedded in _version.py can just fetch the value of these
575	    # keywords. When used from setup.py, we don't want to import _version.py,
576	    # so we do it with a regexp instead. This function is not used from
577	    # _version.py.
578	    keywords = {}
579	    try:
580	        with open(versionfile_abs, "r") as fobj:
581	            for line in fobj:
582	                if line.strip().startswith("git_refnames ="):
583	                    mo = re.search(r'=\s*"(.*)"', line)
584	                    if mo:
585	                        keywords["refnames"] = mo.group(1)
586	                if line.strip().startswith("git_full ="):
587	                    mo = re.search(r'=\s*"(.*)"', line)
588	                    if mo:
589	                        keywords["full"] = mo.group(1)
590	                if line.strip().startswith("git_date ="):
591	                    mo = re.search(r'=\s*"(.*)"', line)
592	                    if mo:
593	                        keywords["date"] = mo.group(1)
594	    except OSError:
595	        pass
596	    return keywords

599	@register_vcs_handler("git", "keywords")
600	def git_versions_from_keywords(keywords, tag_prefix, verbose):
601	    """Get version information from git keywords."""
602	    if "refnames" not in keywords:
603	        raise NotThisMethod("Short version file found")
604	    date = keywords.get("date")
605	    if date is not None:
606	        # Use only the last line.  Previous lines may contain GPG signature
607	        # information.
608	        date = date.splitlines()[-1]
609	
610	        # git-2.2.0 added "%%cI", which expands to an ISO-8601 -compliant
611	        # datestamp. However we prefer "%%ci" (which expands to an "ISO-8601
612	        # -like" string, which we must then edit to make compliant), because
613	        # it's been around since git-1.5.3, and it's too difficult to
614	        # discover which version we're using, or to work around using an
615	        # older one.
616	        date = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
617	    refnames = keywords["refnames"].strip()
618	    if refnames.startswith("$Format"):
619	        if verbose:
620	            print("keywords are unexpanded, not using")
621	        raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
622	    refs = {r.strip() for r in refnames.strip("()").split(",")}
623	    # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
624	    # just "foo-1.0". If we see a "tag: " prefix, prefer those.
625	    TAG = "tag: "
626	    tags = {r[len(TAG):] for r in refs if r.startswith(TAG)}
627	    if not tags:
628	        # Either we're using git < 1.8.3, or there really are no tags. We use
629	        # a heuristic: assume all version tags have a digit. The old git %%d
630	        # expansion behaves like git log --decorate=short and strips out the
631	        # refs/heads/ and refs/tags/ prefixes that would let us distinguish
632	        # between branches and tags. By ignoring refnames without digits, we
633	        # filter out many common branch names like "release" and
634	        # "stabilization", as well as "HEAD" and "master".
635	        tags = {r for r in refs if re.search(r'\d', r)}
636	        if verbose:
637	            print("discarding '%%s', no digits" %% ",".join(refs - tags))
638	    if verbose:
639	        print("likely tags: %%s" %% ",".join(sorted(tags)))
640	    for ref in sorted(tags):
641	        # sorting will prefer e.g. "2.0" over "2.0rc1"
642	        if ref.startswith(tag_prefix):
643	            r = ref[len(tag_prefix):]
644	            # Filter out refs that exactly match prefix or that don't start
645	            # with a number once the prefix is stripped (mostly a concern
646	            # when prefix is '')
647	            if not re.match(r'\d', r):
648	                continue
649	            if verbose:
650	                print("picking %%s" %% r)
651	            return {"version": r,
652	                    "full-revisionid": keywords["full"].strip(),
653	                    "dirty": False, "error": None,
654	                    "date": date}
655	    # no suitable tags, so version is "0+unknown", but full hex is still there
656	    if verbose:
657	        print("no suitable tags, using unknown + full revision id")
658	    return {"version": "0+unknown",
659	            "full-revisionid": keywords["full"].strip(),
660	            "dirty": False, "error": "no suitable tags", "date": None}

663	@register_vcs_handler("git", "pieces_from_vcs")
664	def git_pieces_from_vcs(tag_prefix, root, verbose, runner=run_command):
665	    """Get version from 'git describe' in the root of the source tree.
666	
667	    This only gets called if the git-archive 'subst' keywords were *not*
668	    expanded, and _version.py hasn't already been rewritten with a short
669	    version string, meaning we're inside a checked out source tree.
670	    """
671	    GITS = ["git"]
672	    if sys.platform == "win32":
673	        GITS = ["git.cmd", "git.exe"]
674	
675	    # GIT_DIR can interfere with correct operation of Versioneer.
676	    # It may be intended to be passed to the Versioneer-versioned project,
677	    # but that should not change where we get our version from.
678	    env = os.environ.copy()
679	    env.pop("GIT_DIR", None)
680	    runner = functools.partial(runner, env=env)
681	
682	    _, rc = runner(GITS, ["rev-parse", "--git-dir"], cwd=root,
683	                   hide_stderr=True)
684	    if rc != 0:
685	        if verbose:
686	            print("Directory %%s not under git control" %% root)
687	        raise NotThisMethod("'git rev-parse --git-dir' returned error")
688	
689	    # if there is a tag matching tag_prefix, this yields TAG-NUM-gHEX[-dirty]
690	    # if there isn't one, this yields HEX[-dirty] (no NUM)
691	    describe_out, rc = runner(GITS, [
692	        "describe", "--tags", "--dirty", "--always", "--long",
693	        "--match", f"{tag_prefix}[[:digit:]]*"
694	        ], cwd=root)
695	    # --long was added in git-1.5.5
696	    if describe_out is None:
697	        raise NotThisMethod("'git describe' failed")
698	    describe_out = describe_out.strip()
699	    full_out, rc = runner(GITS, ["rev-parse", "HEAD"], cwd=root)
700	    if full_out is None:
701	        raise NotThisMethod("'git rev-parse' failed")
702	    full_out = full_out.strip()
703	
704	    pieces = {}
705	    pieces["long"] = full_out
706	    pieces["short"] = full_out[:7]  # maybe improved later
707	    pieces["error"] = None
708	
709	    branch_name, rc = runner(GITS, ["rev-parse", "--abbrev-ref", "HEAD"],
710	                             cwd=root)
711	    # --abbrev-ref was added in git-1.6.3
712	    if rc != 0 or branch_name is None:
713	        raise NotThisMethod("'git rev-parse --abbrev-ref' returned error")
714	    branch_name = branch_name.strip()
715	
716	    if branch_name == "HEAD":
717	        # If we aren't exactly on a branch, pick a branch which represents
718	        # the current commit. If all else fails, we are on a branchless
719	        # commit.
720	        branches, rc = runner(GITS, ["branch", "--contains"], cwd=root)
721	        # --contains was added in git-1.5.4
722	        if rc != 0 or branches is None:
723	            raise NotThisMethod("'git branch --contains' returned error")
724	        branches = branches.split("\n")
725	
726	        # Remove the first line if we're running detached
727	        if "(" in branches[0]:
728	            branches.pop(0)
729	
730	        # Strip off the leading "* " from the list of branches.
731	        branches = [branch[2:] for branch in branches]
732	        if "master" in branches:
733	            branch_name = "master"
734	        elif not branches:
735	            branch_name = None
736	        else:
737	            # Pick the first branch that is returned. Good or bad.
738	            branch_name = branches[0]
739	
740	    pieces["branch"] = branch_name
741	
742	    # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
743	    # TAG might have hyphens.
744	    git_describe = describe_out
745	
746	    # look for -dirty suffix
747	    dirty = git_describe.endswith("-dirty")
748	    pieces["dirty"] = dirty
749	    if dirty:
750	        git_describe = git_describe[:git_describe.rindex("-dirty")]
751	
752	    # now we have TAG-NUM-gHEX or HEX
753	
754	    if "-" in git_describe:
755	        # TAG-NUM-gHEX
756	        mo = re.search(r'^(.+)-(\d+)-g([0-9a-f]+)$', git_describe)
757	        if not mo:
758	            # unparsable. Maybe git-describe is misbehaving?
759	            pieces["error"] = ("unable to parse git-describe output: '%%s'"
760	                               %% describe_out)
761	            return pieces
762	
763	        # tag
764	        full_tag = mo.group(1)
765	        if not full_tag.startswith(tag_prefix):
766	            if verbose:
767	                fmt = "tag '%%s' doesn't start with prefix '%%s'"
768	                print(fmt %% (full_tag, tag_prefix))
769	            pieces["error"] = ("tag '%%s' doesn't start with prefix '%%s'"
770	                               %% (full_tag, tag_prefix))
771	            return pieces
772	        pieces["closest-tag"] = full_tag[len(tag_prefix):]
773	
774	        # distance: number of commits since tag
775	        pieces["distance"] = int(mo.group(2))
776	
777	        # commit: short hex revision ID
778	        pieces["short"] = mo.group(3)
779	
780	    else:
781	        # HEX: no tags
782	        pieces["closest-tag"] = None
783	        out, rc = runner(GITS, ["rev-list", "HEAD", "--left-right"], cwd=root)
784	        pieces["distance"] = len(out.split())  # total number of commits
785	
786	    # commit date: see ISO-8601 comment in git_versions_from_keywords()
787	    date = runner(GITS, ["show", "-s", "--format=%%ci", "HEAD"], cwd=root)[0].strip()
788	    # Use only the last line.  Previous lines may contain GPG signature
789	    # information.
790	    date = date.splitlines()[-1]
791	    pieces["date"] = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
792	
793	    return pieces

803	def render_pep440(pieces):
804	    """Build up version string, with post-release "local version identifier".
805	
806	    Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
807	    get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
808	
809	    Exceptions:
810	    1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
811	    """
812	    if pieces["closest-tag"]:
813	        rendered = pieces["closest-tag"]
814	        if pieces["distance"] or pieces["dirty"]:
815	            rendered += plus_or_dot(pieces)
816	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
817	            if pieces["dirty"]:
818	                rendered += ".dirty"
819	    else:
820	        # exception #1
821	        rendered = "0+untagged.%%d.g%%s" %% (pieces["distance"],
822	                                          pieces["short"])
823	        if pieces["dirty"]:
824	            rendered += ".dirty"
825	    return rendered

828	def render_pep440_branch(pieces):
829	    """TAG[[.dev0]+DISTANCE.gHEX[.dirty]] .
830	
831	    The ".dev0" means not master branch. Note that .dev0 sorts backwards
832	    (a feature branch will appear "older" than the master branch).
833	
834	    Exceptions:
835	    1: no tags. 0[.dev0]+untagged.DISTANCE.gHEX[.dirty]
836	    """
837	    if pieces["closest-tag"]:
838	        rendered = pieces["closest-tag"]
839	        if pieces["distance"] or pieces["dirty"]:
840	            if pieces["branch"] != "master":
841	                rendered += ".dev0"
842	            rendered += plus_or_dot(pieces)
843	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
844	            if pieces["dirty"]:
845	                rendered += ".dirty"
846	    else:
847	        # exception #1
848	        rendered = "0"
849	        if pieces["branch"] != "master":
850	            rendered += ".dev0"
851	        rendered += "+untagged.%%d.g%%s" %% (pieces["distance"],
852	                                          pieces["short"])
853	        if pieces["dirty"]:
854	            rendered += ".dirty"
855	    return rendered

858	def pep440_split_post(ver):
859	    """Split pep440 version string at the post-release segment.
860	
861	    Returns the release segments before the post-release and the
862	    post-release version number (or -1 if no post-release segment is present).
863	    """
864	    vc = str.split(ver, ".post")
865	    return vc[0], int(vc[1] or 0) if len(vc) == 2 else None

868	def render_pep440_pre(pieces):
869	    """TAG[.postN.devDISTANCE] -- No -dirty.
870	
871	    Exceptions:
872	    1: no tags. 0.post0.devDISTANCE
873	    """
874	    if pieces["closest-tag"]:
875	        if pieces["distance"]:
876	            # update the post release segment
877	            tag_version, post_version = pep440_split_post(pieces["closest-tag"])
878	            rendered = tag_version
879	            if post_version is not None:
880	                rendered += ".post%%d.dev%%d" %% (post_version + 1, pieces["distance"])
881	            else:
882	                rendered += ".post0.dev%%d" %% (pieces["distance"])
883	        else:
884	            # no commits, use the tag as the version
885	            rendered = pieces["closest-tag"]
886	    else:
887	        # exception #1
888	        rendered = "0.post0.dev%%d" %% pieces["distance"]
889	    return rendered

1010	def render(pieces, style):
1011	    """Render the given version pieces into the requested style."""
1012	    if pieces["error"]:
1013	        return {"version": "unknown",
1014	                "full-revisionid": pieces.get("long"),
1015	                "dirty": None,
1016	                "error": pieces["error"],
1017	                "date": None}
1018	
1019	    if not style or style == "default":
1020	        style = "pep440"  # the default
1021	
1022	    if style == "pep440":
1023	        rendered = render_pep440(pieces)
1024	    elif style == "pep440-branch":
1025	        rendered = render_pep440_branch(pieces)
