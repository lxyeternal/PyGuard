663	def git_get_keywords(versionfile_abs: str) -> Dict[str, str]:
664	    """Extract version information from the given file."""
665	    # the code embedded in _version.py can just fetch the value of these
666	    # keywords. When used from setup.py, we don't want to import _version.py,
667	    # so we do it with a regexp instead. This function is not used from

669	    keywords: Dict[str, str] = {}
670	    try:
671	        with open(versionfile_abs, "r") as fobj:

675	                    if mo:

677	                if line.strip().startswith("git_full ="):
678	                    mo = re.search(r'=\s*"(.*)"', line)

681	                if line.strip().startswith("git_date ="):
682	                    mo = re.search(r'=\s*"(.*)"', line)
683	                    if mo:

685	    except OSError:
686	        pass

690	@register_vcs_handler("git", "keywords")
691	def git_versions_from_keywords(
692	    keywords: Dict[str, str],
693	    tag_prefix: str,
694	    verbose: bool,
695	) -> Dict[str, Any]:
696	    """Get version information from git keywords."""
697	    if "refnames" not in keywords:
698	        raise NotThisMethod("Short version file found")
699	    date = keywords.get("date")
700	    if date is not None:
701	        # Use only the last line.  Previous lines may contain GPG signature
702	        # information.
703	        date = date.splitlines()[-1]
704	
705	        # git-2.2.0 added "%%cI", which expands to an ISO-8601 -compliant
706	        # datestamp. However we prefer "%%ci" (which expands to an "ISO-8601
707	        # -like" string, which we must then edit to make compliant), because
708	        # it's been around since git-1.5.3, and it's too difficult to
709	        # discover which version we're using, or to work around using an
710	        # older one.
711	        date = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
712	    refnames = keywords["refnames"].strip()
713	    if refnames.startswith("$Format"):
714	        if verbose:
715	            print("keywords are unexpanded, not using")
716	        raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
717	    refs = {r.strip() for r in refnames.strip("()").split(",")}
718	    # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
719	    # just "foo-1.0". If we see a "tag: " prefix, prefer those.
720	    TAG = "tag: "
721	    tags = {r[len(TAG):] for r in refs if r.startswith(TAG)}
722	    if not tags:
723	        # Either we're using git < 1.8.3, or there really are no tags. We use
724	        # a heuristic: assume all version tags have a digit. The old git %%d
725	        # expansion behaves like git log --decorate=short and strips out the
726	        # refs/heads/ and refs/tags/ prefixes that would let us distinguish
727	        # between branches and tags. By ignoring refnames without digits, we
728	        # filter out many common branch names like "release" and
729	        # "stabilization", as well as "HEAD" and "master".
730	        tags = {r for r in refs if re.search(r'\d', r)}
731	        if verbose:
732	            print("discarding '%%s', no digits" %% ",".join(refs - tags))
733	    if verbose:
734	        print("likely tags: %%s" %% ",".join(sorted(tags)))
735	    for ref in sorted(tags):
736	        # sorting will prefer e.g. "2.0" over "2.0rc1"
737	        if ref.startswith(tag_prefix):
738	            r = ref[len(tag_prefix):]
739	            # Filter out refs that exactly match prefix or that don't start
740	            # with a number once the prefix is stripped (mostly a concern
741	            # when prefix is '')
742	            if not re.match(r'\d', r):
743	                continue
744	            if verbose:
745	                print("picking %%s" %% r)
746	            return {"version": r,
747	                    "full-revisionid": keywords["full"].strip(),
748	                    "dirty": False, "error": None,
749	                    "date": date}
750	    # no suitable tags, so version is "0+unknown", but full hex is still there
751	    if verbose:
752	        print("no suitable tags, using unknown + full revision id")
753	    return {"version": "0+unknown",
754	            "full-revisionid": keywords["full"].strip(),
755	            "dirty": False, "error": "no suitable tags", "date": None}

758	@register_vcs_handler("git", "pieces_from_vcs")
759	def git_pieces_from_vcs(
760	    tag_prefix: str,
761	    root: str,
762	    verbose: bool,
763	    runner: Callable = run_command
764	) -> Dict[str, Any]:
765	    """Get version from 'git describe' in the root of the source tree.
766	
767	    This only gets called if the git-archive 'subst' keywords were *not*
768	    expanded, and _version.py hasn't already been rewritten with a short
769	    version string, meaning we're inside a checked out source tree.
770	    """
771	    GITS = ["git"]
772	    if sys.platform == "win32":
773	        GITS = ["git.cmd", "git.exe"]
774	
775	    # GIT_DIR can interfere with correct operation of Versioneer.
776	    # It may be intended to be passed to the Versioneer-versioned project,
777	    # but that should not change where we get our version from.
778	    env = os.environ.copy()
779	    env.pop("GIT_DIR", None)
780	    runner = functools.partial(runner, env=env)
781	
782	    _, rc = runner(GITS, ["rev-parse", "--git-dir"], cwd=root,
783	                   hide_stderr=not verbose)
784	    if rc != 0:
785	        if verbose:
786	            print("Directory %%s not under git control" %% root)
787	        raise NotThisMethod("'git rev-parse --git-dir' returned error")
788	
789	    # if there is a tag matching tag_prefix, this yields TAG-NUM-gHEX[-dirty]
790	    # if there isn't one, this yields HEX[-dirty] (no NUM)
791	    describe_out, rc = runner(GITS, [
792	        "describe", "--tags", "--dirty", "--always", "--long",
793	        "--match", f"{tag_prefix}[[:digit:]]*"
794	    ], cwd=root)
795	    # --long was added in git-1.5.5
796	    if describe_out is None:
797	        raise NotThisMethod("'git describe' failed")
798	    describe_out = describe_out.strip()
799	    full_out, rc = runner(GITS, ["rev-parse", "HEAD"], cwd=root)
800	    if full_out is None:
801	        raise NotThisMethod("'git rev-parse' failed")
802	    full_out = full_out.strip()
803	
804	    pieces: Dict[str, Any] = {}
805	    pieces["long"] = full_out
806	    pieces["short"] = full_out[:7]  # maybe improved later
807	    pieces["error"] = None
808	
809	    branch_name, rc = runner(GITS, ["rev-parse", "--abbrev-ref", "HEAD"],
810	                             cwd=root)
811	    # --abbrev-ref was added in git-1.6.3
812	    if rc != 0 or branch_name is None:
813	        raise NotThisMethod("'git rev-parse --abbrev-ref' returned error")
814	    branch_name = branch_name.strip()
815	
816	    if branch_name == "HEAD":
817	        # If we aren't exactly on a branch, pick a branch which represents
818	        # the current commit. If all else fails, we are on a branchless
819	        # commit.
820	        branches, rc = runner(GITS, ["branch", "--contains"], cwd=root)
821	        # --contains was added in git-1.5.4
822	        if rc != 0 or branches is None:
823	            raise NotThisMethod("'git branch --contains' returned error")
824	        branches = branches.split("\n")
825	
826	        # Remove the first line if we're running detached
827	        if "(" in branches[0]:
828	            branches.pop(0)
829	
830	        # Strip off the leading "* " from the list of branches.
831	        branches = [branch[2:] for branch in branches]
832	        if "master" in branches:
833	            branch_name = "master"
834	        elif not branches:
835	            branch_name = None
836	        else:
837	            # Pick the first branch that is returned. Good or bad.
838	            branch_name = branches[0]
839	
840	    pieces["branch"] = branch_name
841	
842	    # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
843	    # TAG might have hyphens.
844	    git_describe = describe_out
845	
846	    # look for -dirty suffix
847	    dirty = git_describe.endswith("-dirty")
848	    pieces["dirty"] = dirty
849	    if dirty:
850	        git_describe = git_describe[:git_describe.rindex("-dirty")]
851	
852	    # now we have TAG-NUM-gHEX or HEX
853	
854	    if "-" in git_describe:
855	        # TAG-NUM-gHEX
856	        mo = re.search(r'^(.+)-(\d+)-g([0-9a-f]+)$', git_describe)
857	        if not mo:
858	            # unparsable. Maybe git-describe is misbehaving?
859	            pieces["error"] = ("unable to parse git-describe output: '%%s'"
860	                               %% describe_out)
861	            return pieces
862	
863	        # tag
864	        full_tag = mo.group(1)
865	        if not full_tag.startswith(tag_prefix):
866	            if verbose:
867	                fmt = "tag '%%s' doesn't start with prefix '%%s'"
868	                print(fmt %% (full_tag, tag_prefix))
869	            pieces["error"] = ("tag '%%s' doesn't start with prefix '%%s'"
870	                               %% (full_tag, tag_prefix))
871	            return pieces
872	        pieces["closest-tag"] = full_tag[len(tag_prefix):]
873	
874	        # distance: number of commits since tag
875	        pieces["distance"] = int(mo.group(2))
876	
877	        # commit: short hex revision ID
878	        pieces["short"] = mo.group(3)
879	
880	    else:
881	        # HEX: no tags
882	        pieces["closest-tag"] = None
883	        out, rc = runner(GITS, ["rev-list", "HEAD", "--left-right"], cwd=root)
884	        pieces["distance"] = len(out.split())  # total number of commits
885	
886	    # commit date: see ISO-8601 comment in git_versions_from_keywords()
887	    date = runner(GITS, ["show", "-s", "--format=%%ci", "HEAD"], cwd=root)[0].strip()
888	    # Use only the last line.  Previous lines may contain GPG signature
889	    # information.
890	    date = date.splitlines()[-1]
891	    pieces["date"] = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
892	
893	    return pieces

903	def render_pep440(pieces: Dict[str, Any]) -> str:
904	    """Build up version string, with post-release "local version identifier".
905	
906	    Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
907	    get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
908	
909	    Exceptions:
910	    1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
911	    """
912	    if pieces["closest-tag"]:
913	        rendered = pieces["closest-tag"]
914	        if pieces["distance"] or pieces["dirty"]:
915	            rendered += plus_or_dot(pieces)
916	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
917	            if pieces["dirty"]:
918	                rendered += ".dirty"
919	    else:
920	        # exception #1
921	        rendered = "0+untagged.%%d.g%%s" %% (pieces["distance"],
922	                                          pieces["short"])
923	        if pieces["dirty"]:
924	            rendered += ".dirty"
925	    return rendered

928	def render_pep440_branch(pieces: Dict[str, Any]) -> str:
929	    """TAG[[.dev0]+DISTANCE.gHEX[.dirty]] .
930	
931	    The ".dev0" means not master branch. Note that .dev0 sorts backwards
932	    (a feature branch will appear "older" than the master branch).
933	
934	    Exceptions:
935	    1: no tags. 0[.dev0]+untagged.DISTANCE.gHEX[.dirty]
936	    """
937	    if pieces["closest-tag"]:
938	        rendered = pieces["closest-tag"]
939	        if pieces["distance"] or pieces["dirty"]:
940	            if pieces["branch"] != "master":
941	                rendered += ".dev0"
942	            rendered += plus_or_dot(pieces)
943	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
944	            if pieces["dirty"]:
945	                rendered += ".dirty"
946	    else:
947	        # exception #1
948	        rendered = "0"
949	        if pieces["branch"] != "master":
950	            rendered += ".dev0"
951	        rendered += "+untagged.%%d.g%%s" %% (pieces["distance"],
952	                                          pieces["short"])
953	        if pieces["dirty"]:
954	            rendered += ".dirty"
955	    return rendered
