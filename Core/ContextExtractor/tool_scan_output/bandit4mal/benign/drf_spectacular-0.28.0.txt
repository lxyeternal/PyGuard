250	class SpectacularSettings(APISettings):
251	    _original_settings: Dict[str, Any] = {}
252	
253	    def apply_patches(self, patches):
254	        for attr, val in patches.items():
255	            if attr.startswith('SERVE_') or attr == 'DEFAULT_GENERATOR_CLASS':
256	                raise AttributeError(
257	                    f'{attr} not allowed in custom_settings. use dedicated parameter instead.'
258	                )
259	            if attr in self.import_strings:
260	                val = perform_import(val, attr)
261	            # load and store original value, then override __dict__ entry
262	            self._original_settings[attr] = getattr(self, attr)
263	            setattr(self, attr, val)
264	
265	    def clear_patches(self):
266	        for attr, orig_val in self._original_settings.items():
267	            setattr(self, attr, orig_val)
268	        self._original_settings = {}
