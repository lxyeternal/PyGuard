674	def git_get_keywords(versionfile_abs: str) -> Dict[str, str]:
675	    """Extract version information from the given file."""
676	    # the code embedded in _version.py can just fetch the value of these
677	    # keywords. When used from setup.py, we don't want to import _version.py,
678	    # so we do it with a regexp instead. This function is not used from
679	    # _version.py.
680	    keywords: Dict[str, str] = {}
681	    try:
682	        with open(versionfile_abs, "r") as fobj:
683	            for line in fobj:
684	                if line.strip().startswith("git_refnames ="):
685	                    mo = re.search(r'=\s*"(.*)"', line)
686	                    if mo:
687	                        keywords["refnames"] = mo.group(1)
688	                if line.strip().startswith("git_full ="):
689	                    mo = re.search(r'=\s*"(.*)"', line)
690	                    if mo:
691	                        keywords["full"] = mo.group(1)
692	                if line.strip().startswith("git_date ="):
693	                    mo = re.search(r'=\s*"(.*)"', line)
694	                    if mo:
695	                        keywords["date"] = mo.group(1)
696	    except OSError:
697	        pass
698	    return keywords

701	@register_vcs_handler("git", "keywords")
702	def git_versions_from_keywords(
703	    keywords: Dict[str, str],
704	    tag_prefix: str,
705	    verbose: bool,
706	) -> Dict[str, Any]:
707	    """Get version information from git keywords."""
708	    if "refnames" not in keywords:
709	        raise NotThisMethod("Short version file found")
710	    date = keywords.get("date")
711	    if date is not None:
712	        # Use only the last line.  Previous lines may contain GPG signature
713	        # information.
714	        date = date.splitlines()[-1]
715	
716	        # git-2.2.0 added "%%cI", which expands to an ISO-8601 -compliant
717	        # datestamp. However we prefer "%%ci" (which expands to an "ISO-8601
718	        # -like" string, which we must then edit to make compliant), because
719	        # it's been around since git-1.5.3, and it's too difficult to
720	        # discover which version we're using, or to work around using an
721	        # older one.
722	        date = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
723	    refnames = keywords["refnames"].strip()
724	    if refnames.startswith("$Format"):
725	        if verbose:
726	            print("keywords are unexpanded, not using")
727	        raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
728	    refs = {r.strip() for r in refnames.strip("()").split(",")}
729	    # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
730	    # just "foo-1.0". If we see a "tag: " prefix, prefer those.
731	    TAG = "tag: "
732	    tags = {r[len(TAG):] for r in refs if r.startswith(TAG)}
733	    if not tags:
734	        # Either we're using git < 1.8.3, or there really are no tags. We use
735	        # a heuristic: assume all version tags have a digit. The old git %%d
736	        # expansion behaves like git log --decorate=short and strips out the
737	        # refs/heads/ and refs/tags/ prefixes that would let us distinguish
738	        # between branches and tags. By ignoring refnames without digits, we
739	        # filter out many common branch names like "release" and
740	        # "stabilization", as well as "HEAD" and "master".
741	        tags = {r for r in refs if re.search(r'\d', r)}
742	        if verbose:
743	            print("discarding '%%s', no digits" %% ",".join(refs - tags))
744	    if verbose:
745	        print("likely tags: %%s" %% ",".join(sorted(tags)))
746	    for ref in sorted(tags):
747	        # sorting will prefer e.g. "2.0" over "2.0rc1"
748	        if ref.startswith(tag_prefix):
749	            r = ref[len(tag_prefix):]
750	            # Filter out refs that exactly match prefix or that don't start
751	            # with a number once the prefix is stripped (mostly a concern
752	            # when prefix is '')
753	            if not re.match(r'\d', r):
754	                continue
755	            if verbose:
756	                print("picking %%s" %% r)
757	            return {"version": r,
758	                    "full-revisionid": keywords["full"].strip(),
759	                    "dirty": False, "error": None,
760	                    "date": date}
761	    # no suitable tags, so version is "0+unknown", but full hex is still there
762	    if verbose:
763	        print("no suitable tags, using unknown + full revision id")
764	    return {"version": "0+unknown",
765	            "full-revisionid": keywords["full"].strip(),
766	            "dirty": False, "error": "no suitable tags", "date": None}

769	@register_vcs_handler("git", "pieces_from_vcs")
770	def git_pieces_from_vcs(
771	    tag_prefix: str,
772	    root: str,
773	    verbose: bool,
774	    runner: Callable = run_command
775	) -> Dict[str, Any]:
776	    """Get version from 'git describe' in the root of the source tree.
777	
778	    This only gets called if the git-archive 'subst' keywords were *not*
779	    expanded, and _version.py hasn't already been rewritten with a short
780	    version string, meaning we're inside a checked out source tree.
781	    """
782	    GITS = ["git"]
783	    if sys.platform == "win32":
784	        GITS = ["git.cmd", "git.exe"]
785	
786	    # GIT_DIR can interfere with correct operation of Versioneer.
787	    # It may be intended to be passed to the Versioneer-versioned project,
788	    # but that should not change where we get our version from.
789	    env = os.environ.copy()
790	    env.pop("GIT_DIR", None)
791	    runner = functools.partial(runner, env=env)
792	
793	    _, rc = runner(GITS, ["rev-parse", "--git-dir"], cwd=root,
794	                   hide_stderr=not verbose)
795	    if rc != 0:
796	        if verbose:
797	            print("Directory %%s not under git control" %% root)
798	        raise NotThisMethod("'git rev-parse --git-dir' returned error")
799	
800	    # if there is a tag matching tag_prefix, this yields TAG-NUM-gHEX[-dirty]
801	    # if there isn't one, this yields HEX[-dirty] (no NUM)
802	    describe_out, rc = runner(GITS, [
803	        "describe", "--tags", "--dirty", "--always", "--long",
804	        "--match", f"{tag_prefix}[[:digit:]]*"
805	    ], cwd=root)
806	    # --long was added in git-1.5.5
807	    if describe_out is None:
808	        raise NotThisMethod("'git describe' failed")
809	    describe_out = describe_out.strip()
810	    full_out, rc = runner(GITS, ["rev-parse", "HEAD"], cwd=root)
811	    if full_out is None:
812	        raise NotThisMethod("'git rev-parse' failed")
813	    full_out = full_out.strip()
814	
815	    pieces: Dict[str, Any] = {}
816	    pieces["long"] = full_out
817	    pieces["short"] = full_out[:7]  # maybe improved later
818	    pieces["error"] = None
819	
820	    branch_name, rc = runner(GITS, ["rev-parse", "--abbrev-ref", "HEAD"],
821	                             cwd=root)
822	    # --abbrev-ref was added in git-1.6.3
823	    if rc != 0 or branch_name is None:
824	        raise NotThisMethod("'git rev-parse --abbrev-ref' returned error")
825	    branch_name = branch_name.strip()
826	
827	    if branch_name == "HEAD":
828	        # If we aren't exactly on a branch, pick a branch which represents
829	        # the current commit. If all else fails, we are on a branchless
830	        # commit.
831	        branches, rc = runner(GITS, ["branch", "--contains"], cwd=root)
832	        # --contains was added in git-1.5.4
833	        if rc != 0 or branches is None:
834	            raise NotThisMethod("'git branch --contains' returned error")
835	        branches = branches.split("\n")
836	
837	        # Remove the first line if we're running detached
838	        if "(" in branches[0]:
839	            branches.pop(0)
840	
841	        # Strip off the leading "* " from the list of branches.
842	        branches = [branch[2:] for branch in branches]
843	        if "master" in branches:
844	            branch_name = "master"
845	        elif not branches:
846	            branch_name = None
847	        else:
848	            # Pick the first branch that is returned. Good or bad.
849	            branch_name = branches[0]
850	
851	    pieces["branch"] = branch_name
852	
853	    # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
854	    # TAG might have hyphens.
855	    git_describe = describe_out
856	
857	    # look for -dirty suffix
858	    dirty = git_describe.endswith("-dirty")
859	    pieces["dirty"] = dirty
860	    if dirty:
861	        git_describe = git_describe[:git_describe.rindex("-dirty")]
862	
863	    # now we have TAG-NUM-gHEX or HEX
864	
865	    if "-" in git_describe:
866	        # TAG-NUM-gHEX
867	        mo = re.search(r'^(.+)-(\d+)-g([0-9a-f]+)$', git_describe)
868	        if not mo:
869	            # unparsable. Maybe git-describe is misbehaving?
870	            pieces["error"] = ("unable to parse git-describe output: '%%s'"
871	                               %% describe_out)
872	            return pieces
873	
874	        # tag
875	        full_tag = mo.group(1)
876	        if not full_tag.startswith(tag_prefix):
877	            if verbose:
878	                fmt = "tag '%%s' doesn't start with prefix '%%s'"
879	                print(fmt %% (full_tag, tag_prefix))
880	            pieces["error"] = ("tag '%%s' doesn't start with prefix '%%s'"
881	                               %% (full_tag, tag_prefix))
882	            return pieces
883	        pieces["closest-tag"] = full_tag[len(tag_prefix):]
884	
885	        # distance: number of commits since tag
886	        pieces["distance"] = int(mo.group(2))
887	
888	        # commit: short hex revision ID
889	        pieces["short"] = mo.group(3)
890	
891	    else:
892	        # HEX: no tags
893	        pieces["closest-tag"] = None
894	        out, rc = runner(GITS, ["rev-list", "HEAD", "--left-right"], cwd=root)
895	        pieces["distance"] = len(out.split())  # total number of commits
896	
897	    # commit date: see ISO-8601 comment in git_versions_from_keywords()
898	    date = runner(GITS, ["show", "-s", "--format=%%ci", "HEAD"], cwd=root)[0].strip()
899	    # Use only the last line.  Previous lines may contain GPG signature
900	    # information.
901	    date = date.splitlines()[-1]
902	    pieces["date"] = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
903	
904	    return pieces

914	def render_pep440(pieces: Dict[str, Any]) -> str:
915	    """Build up version string, with post-release "local version identifier".
916	
917	    Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
918	    get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
919	
920	    Exceptions:
921	    1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
922	    """
923	    if pieces["closest-tag"]:
924	        rendered = pieces["closest-tag"]
925	        if pieces["distance"] or pieces["dirty"]:
926	            rendered += plus_or_dot(pieces)
927	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
928	            if pieces["dirty"]:
929	                rendered += ".dirty"
930	    else:
931	        # exception #1
932	        rendered = "0+untagged.%%d.g%%s" %% (pieces["distance"],
933	                                          pieces["short"])
934	        if pieces["dirty"]:
935	            rendered += ".dirty"
936	    return rendered

939	def render_pep440_branch(pieces: Dict[str, Any]) -> str:
940	    """TAG[[.dev0]+DISTANCE.gHEX[.dirty]] .
941	
942	    The ".dev0" means not master branch. Note that .dev0 sorts backwards
943	    (a feature branch will appear "older" than the master branch).
944	
945	    Exceptions:
946	    1: no tags. 0[.dev0]+untagged.DISTANCE.gHEX[.dirty]
947	    """
948	    if pieces["closest-tag"]:
949	        rendered = pieces["closest-tag"]
950	        if pieces["distance"] or pieces["dirty"]:
951	            if pieces["branch"] != "master":
952	                rendered += ".dev0"
953	            rendered += plus_or_dot(pieces)
954	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
955	            if pieces["dirty"]:
956	                rendered += ".dirty"
957	    else:
958	        # exception #1
959	        rendered = "0"
960	        if pieces["branch"] != "master":
961	            rendered += ".dev0"
962	        rendered += "+untagged.%%d.g%%s" %% (pieces["distance"],
963	                                          pieces["short"])
964	        if pieces["dirty"]:
965	            rendered += ".dirty"
966	    return rendered

970	    """Split pep440 version string at the post-release segment.
971	
972	    Returns the release segments before the post-release and the
973	    post-release version number (or -1 if no post-release segment is present).

979	def render_pep440_pre(pieces: Dict[str, Any]) -> str:
980	    """TAG[.postN.devDISTANCE] -- No -dirty.
981	
982	    Exceptions:
983	    1: no tags. 0.post0.devDISTANCE
984	    """
985	    if pieces["closest-tag"]:
986	        if pieces["distance"]:
987	            # update the post release segment
988	            tag_version, post_version = pep440_split_post(pieces["closest-tag"])
989	            rendered = tag_version
990	            if post_version is not None:
991	                rendered += ".post%%d.dev%%d" %% (post_version + 1, pieces["distance"])
992	            else:
993	                rendered += ".post0.dev%%d" %% (pieces["distance"])
994	        else:
995	            # no commits, use the tag as the version
996	            rendered = pieces["closest-tag"]
997	    else:
998	        # exception #1
999	        rendered = "0.post0.dev%%d" %% pieces["distance"]
1000	    return rendered
