596	@register_vcs_handler("git", "keywords")
597	def git_versions_from_keywords(keywords, tag_prefix, verbose):
598	    if not keywords:
599	        raise NotThisMethod("no keywords at all, weird")
600	    refnames = keywords["refnames"].strip()
601	    if refnames.startswith("$Format"):
602	        if verbose:
603	            print("keywords are unexpanded, not using")
604	        raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
605	    refs = set([r.strip() for r in refnames.strip("()").split(",")])
606	    # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
607	    # just "foo-1.0". If we see a "tag: " prefix, prefer those.
608	    TAG = "tag: "
609	    tags = set([r[len(TAG):] for r in refs if r.startswith(TAG)])
610	    if not tags:
611	        # Either we're using git < 1.8.3, or there really are no tags. We use
612	        # a heuristic: assume all version tags have a digit. The old git %%d
613	        # expansion behaves like git log --decorate=short and strips out the
614	        # refs/heads/ and refs/tags/ prefixes that would let us distinguish
615	        # between branches and tags. By ignoring refnames without digits, we
616	        # filter out many common branch names like "release" and
617	        # "stabilization", as well as "HEAD" and "master".
618	        tags = set([r for r in refs if re.search(r'\d', r)])
619	        if verbose:
620	            print("discarding '%%s', no digits" %% ",".join(refs-tags))
621	    if verbose:
622	        print("likely tags: %%s" %% ",".join(sorted(tags)))
623	    for ref in sorted(tags):
624	        # sorting will prefer e.g. "2.0" over "2.0rc1"
625	        if ref.startswith(tag_prefix):
626	            r = ref[len(tag_prefix):]
627	            if verbose:
628	                print("picking %%s" %% r)
629	            return {"version": r,
630	                    "full-revisionid": keywords["full"].strip(),
631	                    "dirty": False, "error": None
632	                    }
633	    # no suitable tags, so version is "0+unknown", but full hex is still there
634	    if verbose:
635	        print("no suitable tags, using unknown + full revision id")
636	    return {"version": "0+unknown",
637	            "full-revisionid": keywords["full"].strip(),
638	            "dirty": False, "error": "no suitable tags"}

641	@register_vcs_handler("git", "pieces_from_vcs")
642	def git_pieces_from_vcs(tag_prefix, root, verbose, run_command=run_command):
643	    # this runs 'git' from the root of the source tree. This only gets called
644	    # if the git-archive 'subst' keywords were *not* expanded, and
645	    # _version.py hasn't already been rewritten with a short version string,
646	    # meaning we're inside a checked out source tree.
647	
648	    if not os.path.exists(os.path.join(root, ".git")):
649	        if verbose:
650	            print("no .git in %%s" %% root)
651	        raise NotThisMethod("no .git directory")
652	
653	    GITS = ["git"]
654	    if sys.platform == "win32":
655	        GITS = ["git.cmd", "git.exe"]
656	    # if there is a tag, this yields TAG-NUM-gHEX[-dirty]
657	    # if there are no tags, this yields HEX[-dirty] (no NUM)
658	    describe_out = run_command(GITS, ["describe", "--tags", "--dirty",
659	                                      "--always", "--long"],
660	                               cwd=root)
661	    # --long was added in git-1.5.5
662	    if describe_out is None:
663	        raise NotThisMethod("'git describe' failed")
664	    describe_out = describe_out.strip()
665	    full_out = run_command(GITS, ["rev-parse", "HEAD"], cwd=root)
666	    if full_out is None:
667	        raise NotThisMethod("'git rev-parse' failed")
668	    full_out = full_out.strip()
669	
670	    pieces = {}
671	    pieces["long"] = full_out
672	    pieces["short"] = full_out[:7]  # maybe improved later
673	    pieces["error"] = None
674	
675	    # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
676	    # TAG might have hyphens.
677	    git_describe = describe_out
678	
679	    # look for -dirty suffix
680	    dirty = git_describe.endswith("-dirty")
681	    pieces["dirty"] = dirty
682	    if dirty:
683	        git_describe = git_describe[:git_describe.rindex("-dirty")]
684	
685	    # now we have TAG-NUM-gHEX or HEX
686	
687	    if "-" in git_describe:
688	        # TAG-NUM-gHEX
689	        mo = re.search(r'^(.+)-(\d+)-g([0-9a-f]+)$', git_describe)
690	        if not mo:
691	            # unparseable. Maybe git-describe is misbehaving?
692	            pieces["error"] = ("unable to parse git-describe output: '%%s'"
693	                               %% describe_out)
694	            return pieces
695	
696	        # tag
697	        full_tag = mo.group(1)
698	        if not full_tag.startswith(tag_prefix):
699	            if verbose:
700	                fmt = "tag '%%s' doesn't start with prefix '%%s'"
701	                print(fmt %% (full_tag, tag_prefix))
702	            pieces["error"] = ("tag '%%s' doesn't start with prefix '%%s'"
703	                               %% (full_tag, tag_prefix))
704	            return pieces
705	        pieces["closest-tag"] = full_tag[len(tag_prefix):]
706	
707	        # distance: number of commits since tag
708	        pieces["distance"] = int(mo.group(2))
709	
710	        # commit: short hex revision ID
711	        pieces["short"] = mo.group(3)
712	
713	    else:
714	        # HEX: no tags
715	        pieces["closest-tag"] = None
716	        count_out = run_command(GITS, ["rev-list", "HEAD", "--count"],
717	                                cwd=root)
718	        pieces["distance"] = int(count_out)  # total number of commits
719	
720	    return pieces

729	def render_pep440(pieces):
730	    # now build up version string, with post-release "local version
731	    # identifier". Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
732	    # get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
733	
734	    # exceptions:
735	    # 1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
736	
737	    if pieces["closest-tag"]:
738	        rendered = pieces["closest-tag"]
739	        if pieces["distance"] or pieces["dirty"]:
740	            rendered += plus_or_dot(pieces)
741	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
742	            if pieces["dirty"]:
743	                rendered += ".dirty"
744	    else:
745	        # exception #1
746	        rendered = "0+untagged.%%d.g%%s" %% (pieces["distance"],
747	                                          pieces["short"])
748	        if pieces["dirty"]:
749	            rendered += ".dirty"
750	    return rendered

753	def render_pep440_pre(pieces):
754	    # TAG[.post.devDISTANCE] . No -dirty
755	
756	    # exceptions:
757	    # 1: no tags. 0.post.devDISTANCE
758	
759	    if pieces["closest-tag"]:
760	        rendered = pieces["closest-tag"]
761	        if pieces["distance"]:
762	            rendered += ".post.dev%%d" %% pieces["distance"]
763	    else:
764	        # exception #1
765	        rendered = "0.post.dev%%d" %% pieces["distance"]
766	    return rendered

769	def render_pep440_post(pieces):
770	    # TAG[.postDISTANCE[.dev0]+gHEX] . The ".dev0" means dirty. Note that
771	    # .dev0 sorts backwards (a dirty tree will appear "older" than the
772	    # corresponding clean one), but you shouldn't be releasing software with
773	    # -dirty anyways.
774	
775	    # exceptions:
776	    # 1: no tags. 0.postDISTANCE[.dev0]
777	
778	    if pieces["closest-tag"]:
779	        rendered = pieces["closest-tag"]
780	        if pieces["distance"] or pieces["dirty"]:
781	            rendered += ".post%%d" %% pieces["distance"]
782	            if pieces["dirty"]:
783	                rendered += ".dev0"
784	            rendered += plus_or_dot(pieces)
785	            rendered += "g%%s" %% pieces["short"]
786	    else:
787	        # exception #1
788	        rendered = "0.post%%d" %% pieces["distance"]
789	        if pieces["dirty"]:
790	            rendered += ".dev0"
791	        rendered += "+g%%s" %% pieces["short"]
792	    return rendered

852	def render(pieces, style):
853	    if pieces["error"]:
854	        return {"version": "unknown",
855	                "full-revisionid": pieces.get("long"),
856	                "dirty": None,
857	                "error": pieces["error"]}
858

860	        style = "pep440"  # the default
861	
862	    if style == "pep440":
863	        rendered = render_pep440(pieces)
864	    elif style == "pep440-pre":
865	        rendered = render_pep440_pre(pieces)

867	        rendered = render_pep440_post(pieces)
868	    elif style == "pep440-old":
869	        rendered = render_pep440_old(pieces)
870	    elif style == "git-describe":

872	    elif style == "git-describe-long":
873	        rendered = render_git_describe_long(pieces)
874	    else:
875	        raise ValueError("unknown style '%%s'" %% style)
876	
877	    return {"version": rendered, "full-revisionid": pieces["long"],
878	            "dirty": pieces["dirty"], "error": None}
