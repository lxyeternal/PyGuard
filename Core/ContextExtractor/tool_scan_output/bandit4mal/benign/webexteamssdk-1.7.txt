503	def run_command(commands, args, cwd=None, verbose=False, hide_stderr=False,
504	                env=None):
505	    """Call the given command(s)."""
506	    assert isinstance(commands, list)
507	    p = None
508	    for c in commands:
509	        try:
510	            dispcmd = str([c] + args)
511	            # remember shell=False, so use git.cmd on windows, not just git
512	            p = subprocess.Popen([c] + args, cwd=cwd, env=env,

517	        except EnvironmentError:

594	@register_vcs_handler("git", "keywords")
595	def git_versions_from_keywords(keywords, tag_prefix, verbose):
596	    """Get version information from git keywords."""
597	    if not keywords:
598	        raise NotThisMethod("no keywords at all, weird")
599	    date = keywords.get("date")
600	    if date is not None:
601	        # git-2.2.0 added "%%cI", which expands to an ISO-8601 -compliant
602	        # datestamp. However we prefer "%%ci" (which expands to an "ISO-8601
603	        # -like" string, which we must then edit to make compliant), because
604	        # it's been around since git-1.5.3, and it's too difficult to
605	        # discover which version we're using, or to work around using an
606	        # older one.
607	        date = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
608	    refnames = keywords["refnames"].strip()
609	    if refnames.startswith("$Format"):
610	        if verbose:
611	            print("keywords are unexpanded, not using")
612	        raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
613	    refs = set([r.strip() for r in refnames.strip("()").split(",")])
614	    # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
615	    # just "foo-1.0". If we see a "tag: " prefix, prefer those.
616	    TAG = "tag: "
617	    tags = set([r[len(TAG):] for r in refs if r.startswith(TAG)])
618	    if not tags:
619	        # Either we're using git < 1.8.3, or there really are no tags. We use
620	        # a heuristic: assume all version tags have a digit. The old git %%d
621	        # expansion behaves like git log --decorate=short and strips out the
622	        # refs/heads/ and refs/tags/ prefixes that would let us distinguish
623	        # between branches and tags. By ignoring refnames without digits, we
624	        # filter out many common branch names like "release" and
625	        # "stabilization", as well as "HEAD" and "master".
626	        tags = set([r for r in refs if re.search(r'\d', r)])
627	        if verbose:
628	            print("discarding '%%s', no digits" %% ",".join(refs - tags))
629	    if verbose:
630	        print("likely tags: %%s" %% ",".join(sorted(tags)))
631	    for ref in sorted(tags):
632	        # sorting will prefer e.g. "2.0" over "2.0rc1"
633	        if ref.startswith(tag_prefix):
634	            r = ref[len(tag_prefix):]
635	            if verbose:
636	                print("picking %%s" %% r)
637	            return {"version": r,
638	                    "full-revisionid": keywords["full"].strip(),
639	                    "dirty": False, "error": None,
640	                    "date": date}
641	    # no suitable tags, so version is "0+unknown", but full hex is still there
642	    if verbose:
643	        print("no suitable tags, using unknown + full revision id")
644	    return {"version": "0+unknown",
645	            "full-revisionid": keywords["full"].strip(),
646	            "dirty": False, "error": "no suitable tags", "date": None}

649	@register_vcs_handler("git", "pieces_from_vcs")
650	def git_pieces_from_vcs(tag_prefix, root, verbose, run_command=run_command):
651	    """Get version from 'git describe' in the root of the source tree.
652	
653	    This only gets called if the git-archive 'subst' keywords were *not*
654	    expanded, and _version.py hasn't already been rewritten with a short
655	    version string, meaning we're inside a checked out source tree.
656	    """
657	    GITS = ["git"]
658	    if sys.platform == "win32":
659	        GITS = ["git.cmd", "git.exe"]
660	
661	    out, rc = run_command(GITS, ["rev-parse", "--git-dir"], cwd=root,
662	                          hide_stderr=True)
663	    if rc != 0:
664	        if verbose:
665	            print("Directory %%s not under git control" %% root)
666	        raise NotThisMethod("'git rev-parse --git-dir' returned error")
667	
668	    # if there is a tag matching tag_prefix, this yields TAG-NUM-gHEX[-dirty]
669	    # if there isn't one, this yields HEX[-dirty] (no NUM)
670	    describe_out, rc = run_command(GITS, ["describe", "--tags", "--dirty",
671	                                          "--always", "--long",
672	                                          "--match", "%%s*" %% tag_prefix],
673	                                   cwd=root)
674	    # --long was added in git-1.5.5
675	    if describe_out is None:
676	        raise NotThisMethod("'git describe' failed")
677	    describe_out = describe_out.strip()
678	    full_out, rc = run_command(GITS, ["rev-parse", "HEAD"], cwd=root)
679	    if full_out is None:
680	        raise NotThisMethod("'git rev-parse' failed")
681	    full_out = full_out.strip()
682	
683	    pieces = {}
684	    pieces["long"] = full_out
685	    pieces["short"] = full_out[:7]  # maybe improved later
686	    pieces["error"] = None
687	
688	    # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
689	    # TAG might have hyphens.
690	    git_describe = describe_out
691	
692	    # look for -dirty suffix
693	    dirty = git_describe.endswith("-dirty")
694	    pieces["dirty"] = dirty
695	    if dirty:
696	        git_describe = git_describe[:git_describe.rindex("-dirty")]
697	
698	    # now we have TAG-NUM-gHEX or HEX
699	
700	    if "-" in git_describe:
701	        # TAG-NUM-gHEX
702	        mo = re.search(r'^(.+)-(\d+)-g([0-9a-f]+)$', git_describe)
703	        if not mo:
704	            # unparseable. Maybe git-describe is misbehaving?
705	            pieces["error"] = ("unable to parse git-describe output: '%%s'"
706	                               %% describe_out)
707	            return pieces
708	
709	        # tag
710	        full_tag = mo.group(1)
711	        if not full_tag.startswith(tag_prefix):
712	            if verbose:
713	                fmt = "tag '%%s' doesn't start with prefix '%%s'"
714	                print(fmt %% (full_tag, tag_prefix))
715	            pieces["error"] = ("tag '%%s' doesn't start with prefix '%%s'"
716	                               %% (full_tag, tag_prefix))
717	            return pieces
718	        pieces["closest-tag"] = full_tag[len(tag_prefix):]
719	
720	        # distance: number of commits since tag
721	        pieces["distance"] = int(mo.group(2))
722	
723	        # commit: short hex revision ID
724	        pieces["short"] = mo.group(3)
725	
726	    else:
727	        # HEX: no tags
728	        pieces["closest-tag"] = None
729	        count_out, rc = run_command(GITS, ["rev-list", "HEAD", "--count"],
730	                                    cwd=root)
731	        pieces["distance"] = int(count_out)  # total number of commits
732	
733	    # commit date: see ISO-8601 comment in git_versions_from_keywords()
734	    date = run_command(GITS, ["show", "-s", "--format=%%ci", "HEAD"],
735	                       cwd=root)[0].strip()
736	    pieces["date"] = date.strip().replace(" ", "T", 1).replace(" ", "", 1)
737	
738	    return pieces

748	def render_pep440(pieces):
749	    """Build up version string, with post-release "local version identifier".
750	
751	    Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
752	    get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
753	
754	    Exceptions:
755	    1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
756	    """
757	    if pieces["closest-tag"]:
758	        rendered = pieces["closest-tag"]
759	        if pieces["distance"] or pieces["dirty"]:
760	            rendered += plus_or_dot(pieces)
761	            rendered += "%%d.g%%s" %% (pieces["distance"], pieces["short"])
762	            if pieces["dirty"]:
763	                rendered += ".dirty"
764	    else:
765	        # exception #1
766	        rendered = "0+untagged.%%d.g%%s" %% (pieces["distance"],
767	                                          pieces["short"])
768	        if pieces["dirty"]:
769	            rendered += ".dirty"
770	    return rendered

790	    """TAG[.postDISTANCE[.dev0]+gHEX] .
791	
792	    The ".dev0" means dirty. Note that .dev0 sorts backwards
793	    (a dirty tree will appear "older" than the corresponding clean one),
794	    but you shouldn't be releasing software with -dirty anyways.
795	
796	    Exceptions:
