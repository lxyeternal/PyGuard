223	def _load_config(config_path, raise_error=False):
224	    """Safely load a config file."""
225	    with open(config_path) as fid:
226	        try:
227	            config = json.load(fid)
228	        except ValueError:
229	            # No JSON object could be decoded --> corrupt file?
230	            msg = (
231	                f"The MNE-Python config file ({config_path}) is not a valid JSON "
232	                "file and might be corrupted"
233	            )
234	            if raise_error:
235	                raise RuntimeError(msg)
236	            warn(msg)
237	            config = dict()
238	    return config
